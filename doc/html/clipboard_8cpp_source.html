00036         QMap<Node*,int> ids;
00037         int next_id = 1;
00038         foreach(Node* n, insert_nodes)
00039         {
00040             nodes.push_back(ClipboardNode(next_id,n->pos()-center));

00046         {

00050                 int id2 = ids.value(e->vertex2(),0);
